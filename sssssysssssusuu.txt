local IsCollapsed = false
local MainFrame = nil
local OriginalSize = nil
local OriginalPosition = nil
local CollapsedSize = UDim2.new(0, 70, 0, 70)

local function SetupCollapseSystem()
    task.wait(0.5)
    
    MainFrame = CoreGui:FindFirstChild("Fluent") and CoreGui.Fluent:FindFirstChild("Main")
    if not MainFrame then 
        warn("ไม่พบ Main Frame")
        return 
    end
    
    OriginalSize = MainFrame.Size
    OriginalPosition = MainFrame.Position
    
    -- สร้างปุ่มพับเก็บสวยๆ
    local CollapseButton = Instance.new("ImageButton")
    CollapseButton.Name = "CollapseButton"
    CollapseButton.Parent = MainFrame
    CollapseButton.AnchorPoint = Vector2.new(1, 0)
    CollapseButton.Position = UDim2.new(1, -10, 0, 10)
    CollapseButton.Size = UDim2.new(0, 32, 0, 32)
    CollapseButton.BackgroundColor3 = Color3.fromRGB(25, 25, 30)
    CollapseButton.BackgroundTransparency = 0.2
    CollapseButton.Image = "rbxassetid://7733964370"
    CollapseButton.ImageColor3 = Color3.fromRGB(255, 255, 255)
    CollapseButton.ZIndex = 1000
    
    local CollapseCorner = Instance.new("UICorner")
    CollapseCorner.CornerRadius = UDim.new(0, 10)
    CollapseCorner.Parent = CollapseButton
    
    local CollapseStroke = Instance.new("UIStroke")
    CollapseStroke.Thickness = 2
    CollapseStroke.Color = Color3.fromRGB(0, 230, 255)
    CollapseStroke.Transparency = 0.6
    CollapseStroke.Parent = CollapseButton
    
    local CollapseGlow = Instance.new("UIGradient")
    CollapseGlow.Color = ColorSequence.new{
        ColorSequenceKeypoint.new(0, Color3.fromRGB(0, 230, 255)),
        ColorSequenceKeypoint.new(1, Color3.fromRGB(255, 0, 230))
    }
    CollapseGlow.Parent = CollapseStroke
    
    -- เอฟเฟกต์ Hover
    CollapseButton.MouseEnter:Connect(function()
        TweenService:Create(CollapseButton, TweenInfo.new(0.2), {
            BackgroundTransparency = 0,
            Size = UDim2.new(0, 36, 0, 36)
        }):Play()
        TweenService:Create(CollapseStroke, TweenInfo.new(0.2), {Transparency = 0.2}):Play()
    end)
    
    CollapseButton.MouseLeave:Connect(function()
        TweenService:Create(CollapseButton, TweenInfo.new(0.2), {
            BackgroundTransparency = 0.2,
            Size = UDim2.new(0, 32, 0, 32)
        }):Play()
        TweenService:Create(CollapseStroke, TweenInfo.new(0.2), {Transparency = 0.6}):Play()
    end)
    
    -- ฟังก์ชันพับเก็บ/ขยาย
    local function ToggleCollapse()
        IsCollapsed = not IsCollapsed
        
        if IsCollapsed then
            -- === พับเก็บ ===
            for _, child in pairs(MainFrame:GetChildren()) do
                if child ~= CollapseButton and child:IsA("GuiObject") and 
                   child.Name ~= "UICorner" and child.Name ~= "UIStroke" then
                    TweenService:Create(child, TweenInfo.new(0.25), {GroupTransparency = 1}):Play()
                end
            end
            
            task.wait(0.15)
            
            TweenService:Create(MainFrame, TweenInfo.new(0.7, Enum.EasingStyle.Back, Enum.EasingDirection.InOut), {
                Size = CollapsedSize,
                Position = UDim2.new(1, -80, 0, 10)
            }):Play()
            
            TweenService:Create(CollapseButton, TweenInfo.new(0.5, Enum.EasingStyle.Back), {Rotation = 180}):Play()
            
            TweenService:Create(CollapseStroke, TweenInfo.new(0.4), {Thickness = 4, Transparency = 0}):Play()
            task.wait(0.4)
            TweenService:Create(CollapseStroke, TweenInfo.new(0.3), {Thickness = 2, Transparency = 0.6}):Play()
            
        else
            -- === ขยายกลับ ===
            TweenService:Create(MainFrame, TweenInfo.new(0.7, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
                Size = OriginalSize,
                Position = OriginalPosition
            }):Play()
            
            TweenService:Create(CollapseButton, TweenInfo.new(0.5, Enum.EasingStyle.Back), {Rotation = 0}):Play()
            
            task.wait(0.4)
            
            for _, child in pairs(MainFrame:GetChildren()) do
                if child ~= CollapseButton and child:IsA("GuiObject") then
                    TweenService:Create(child, TweenInfo.new(0.35, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {GroupTransparency = 0}):Play()
                end
            end
        end
        
        -- เอฟเฟกต์คลิก
        TweenService:Create(CollapseButton, TweenInfo.new(0.1), {Size = UDim2.new(0, 28, 0, 28)}):Play()
        task.wait(0.1)
        TweenService:Create(CollapseButton, TweenInfo.new(0.15, Enum.EasingStyle.Back), {Size = UDim2.new(0, 32, 0, 32)}):Play()
    end
    
    CollapseButton.MouseButton1Click:Connect(ToggleCollapse)
    
    -- ดับเบิลคลิกที่แผงเพื่อพับเก็บ
    local lastClickTime = 0
    MainFrame.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            local currentTime = tick()
            if currentTime - lastClickTime < 0.35 then
                ToggleCollapse()
            end
            lastClickTime = currentTime
        end
    end)
end

task.spawn(SetupCollapseSystem)

local function EnableResize()
    task.wait(0.3)
    local MainFrame = CoreGui.Fluent.Main
    local ResizeBtn = Instance.new("ImageButton")
    
    ResizeBtn.Name = "ResizeHandle"
    ResizeBtn.Parent = MainFrame
    ResizeBtn.Position = UDim2.new(1, -15, 1, -15)
    ResizeBtn.Size = UDim2.new(0, 15, 0, 15)
    ResizeBtn.BackgroundTransparency = 1
    ResizeBtn.Image = "rbxassetid://10615415059"
    ResizeBtn.ImageColor3 = Color3.fromRGB(255, 255, 255)
    ResizeBtn.ImageTransparency = 0.5

    local UserInputService = game:GetService("UserInputService")
    local dragging, dragStart, startSize = false, nil, nil

    ResizeBtn.MouseButton1Down:Connect(function()
        dragging = true
        dragStart = UserInputService:GetMouseLocation()
        startSize = MainFrame.Size
        OriginalSize = startSize
    end)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
            local diff = UserInputService:GetMouseLocation() - dragStart
            MainFrame.Size = UDim2.new(0, math.max(450, startSize.X.Offset + diff.X), 0, math.max(300, startSize.Y.Offset + diff.Y))
        end
    end)

    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = false
        end
    end)
end

task.spawn(EnableResize)

local function GetPlayerList()
    local list = {}
    for _, p in pairs(game:GetService("Players"):GetPlayers()) do
        if p ~= game:GetService("Players").LocalPlayer then
            table.insert(list, p.DisplayName .. " (@" .. p.Name .. ")")
        end
    end
    return list
end

local function CreateESP(plr)
    local Billboard = Instance.new("BillboardGui")
    local TextLabel = Instance.new("TextLabel")
    Billboard.Name = "BOSS69_ESP"
    Billboard.Size = UDim2.new(0, 100, 0, 30)
    Billboard.AlwaysOnTop = true
    Billboard.ExtentsOffset = Vector3.new(0, 3, 0)
    Billboard.Parent = CoreGui
    
    TextLabel.Parent = Billboard
    TextLabel.BackgroundTransparency = 1
    TextLabel.Size = UDim2.new(1, 0, 1, 0)
    TextLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
    TextLabel.TextSize = 10
    
    ESP_Storage[plr] = Billboard
    
    local Connection
    Connection = game:GetService("RunService").Heartbeat:Connect(function()
        if ESP_Enabled and plr.Character and plr.Character:FindFirstChild("Head") then
            local head = plr.Character.Head
            local hum = plr.Character:FindFirstChildOfClass("Humanoid")
            local lpChar = game.Players.LocalPlayer.Character
            local dist = 0
            if lpChar and lpChar:FindFirstChild("Head") then
                dist = math.floor((head.Position - lpChar.Head.Position).Magnitude)
            end
            
            Billboard.Adornee = head
            Billboard.Enabled = true
            TextLabel.Text = string.format("%s\nHP: %d | Dist: %d", plr.DisplayName, (hum and hum.Health or 0), dist)
        else
            Billboard.Enabled = false
            if not plr.Parent then
                Billboard:Destroy()
                Connection:Disconnect()
                ESP_Storage[plr] = nil
            end
        end
    end)
end
